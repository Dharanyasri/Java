import java.util.Scanner;
public class Main{
 public static void main(String args[]){
 Scanner sc = new Scanner(System.in);
 int r = sc.nextInt();
 int h = sc.nextInt();
 int n = sc.nextInt();
 double x = sc.nextInt();

 double volume = 3.14*r*r*h;
 double water = n*h;

 if(water>=volume){
 System.out.printf("The tank can be filled
within " +x+ " hours");
 }
 else{
 System.out.printf("The tank cannot be
filled within " +x+ " hours");
 }


 }
}
