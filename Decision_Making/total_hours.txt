import java.util.Scanner;
public class Main{
 public static void main(String args[]){
 Scanner sc = new Scanner(System.in);
 int hours = sc.nextInt();
 int min = sc.nextInt();
 int sec = sc.nextInt();

 hours = hours+min/60;
 min = min+sec/60;
 sec = sec%60;

 System.out.println("Total Number of hours is
" +hours);
 System.out.println("Total Number of minutes
is " +min);
 System.out.println("Total Number of seconds
is " +sec);
 }
}
