 import java.util.Scanner;
 public class Main{
 public static void main(String args[]){
 Scanner sc = new Scanner(System.in);
 int n = sc.nextInt();
 int fact = 1;
 int i=1;
 while(fact<n){
 i++;
 fact = fact*i;
 }
 if(fact==n){
 System.out.println("Yes");
 }
 else{
 System.out.println("No");
 }
 }
}
