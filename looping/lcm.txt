 import java.util.Scanner;
 public class Main{
  public static void main(String args[]){
    Scanner sc = new Scanner(System.in);
    
    int a = sc.nextInt();
    int b = sc.nextInt();
    int c = sc.nextInt();
    
    int max=a;
    if(b>max){
      max=b;
    }
    if(c>max){
      max=c;
    }
    
    int lcm=max;
    while(true){
      if(lcm%a==0&&lcm%b==0&&lcm%c==0){
        System.out.println(lcm);
        break;
      }
    lcm++;
    }
  }
 }
