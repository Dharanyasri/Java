import java.util.Scanner;
 public class Main{
  public static void main(String args[]){
    Scanner sc = new Scanner(System.in);
    
    int a=sc.nextInt();
    int b=sc.nextInt();
    int c=sc.nextInt();
    
    int min = a;
    if(b<min){
      min=b;
    }
    if(c<min){
      min=c;
    }
    
    int gcf=1;
    for(int i=1; i<=min; i++){
      if(a%i==0&&b%i==0&&c%i==0){
        gcf=i;
      }
    }
    System.out.println(gcf);
  }
 }
